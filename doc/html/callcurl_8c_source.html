26 static size_t WriteMemoryCallback(void *contents, size_t size, size_t nmemb, void *userp)
27 {
28  size_t realsize = size * nmemb;
29  struct MemoryStruct *mem = (struct MemoryStruct *) userp;
30  mem->memory = realloc(mem->memory, mem->size + realsize + 1);
31  if (mem->memory == NULL)
32  {
33  printf("not enough memory (realloc returned NULL)\n");
34  return 0;
35  }
36  memcpy(&(mem->memory[mem->size]), contents, realsize);
37  mem->size += realsize;
38  mem->memory[mem->size] = 0;
39  if (verbose)
40  {
41  printf("Memory size: %d\n", mem->size);
42  printf("Memory value: %s\n", mem->memory);
43  }
44  return realsize;
45 }
