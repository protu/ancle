30 static char *
31 docToChar (xmlDocPtr doc)
32 {
33  char *request = NULL;
34  xmlBufferPtr buf = xmlBufferCreate ();
35  if (buf)
36  {
37  xmlSaveCtxtPtr ctx = xmlSaveToBuffer (buf, "UTF-8",
38  XML_SAVE_FORMAT |
39  XML_SAVE_NO_DECL);
40  if (ctx)
41  {
42  xmlSaveDoc (ctx, doc);
43  xmlSaveClose (ctx);
44  }
45  request = malloc (strlen ((const char *) xmlBufferContent (buf)) + 1);
46  strcpy (request, (const char *) xmlBufferContent (buf));
47  xmlBufferFree (buf);
48  }
49  else
50  {
51  fprintf (stderr, "Error creating xml buffer\n");
52  }
53 
54  xmlFreeDoc (doc);
55  xmlCleanupParser ();
56  xmlMemoryDump ();
57  return (request);
58 }

105  const char *devParams[DEV_PARAM_SIZE] =
106  { "OUI", "SerialNumber", "ProductClass",
107  "Description"
108  };
109  for (i = 0; i < DEV_PARAM_SIZE; i++)
110  {
111  selectfield =
112  xmlNewChild (selectfields, NULL, BAD_CAST "SelectField", NULL);
113  xmlNewChild (selectfield, NULL, BAD_CAST "Type", BAD_CAST "Device");
114  xmlNewChild (selectfield, NULL, BAD_CAST "Name", BAD_CAST devParams[i]);
115  }
116 
117  // Select fiels to search

163 static xmlDocPtr
164 soapStart ()
165 {
166  xmlDocPtr doc = NULL;
167  xmlNodePtr envelope = NULL, body = NULL;
168 
169  doc = xmlNewDoc (BAD_CAST "1.0");
170  envelope = xmlNewNode (NULL, BAD_CAST "Envelope");
171 
172  xmlNsPtr
173  nssoapenv =
174  xmlNewNs (envelope, BAD_CAST "http://schemas.xmlsoap.org/soap/envelope/",
175  BAD_CAST "soapenv");
176  xmlSetNs (envelope, nssoapenv);
177 
178  xmlDocSetRootElement (doc, envelope);
179  xmlNewChild (envelope, NULL, BAD_CAST "Header", NULL);
180  body = xmlNewNode (nssoapenv, BAD_CAST "Body");
181  xmlAddChild (envelope, body);
182 
183  return doc;
184 }
